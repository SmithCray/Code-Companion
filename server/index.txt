const { auth } = require("express-openid-connect");
require("dotenv").config();

app.use(
  auth({
    issuerBaseURL: process.env.ISSUER_BASE_URL,
    baseURL: process.env.CLIENT_ID,
    clientID: process.env.BASE_URL,
    secret: process.env.SECRET,
    idpLogout: true,
  })
);
